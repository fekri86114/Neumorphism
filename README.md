# Dependency

`build.gradle(:app)`:

    def latest_version = "0.3.2"
    implementation "com.github.fornewid:neumorphism:$latest_version"
    
`settings.gradle`:
    
    ...
    dependencyResolutionManagement {
        ...
        repositories {
            ...
            maven { url "https://jitpack.io" }
        }
    }
